# 前后端数据接口定义
 
## 获取图片列表

url: /index-images

type: POST

input: 无

output:

    {
        code: 1,   // 1 成功，0 失败
        message: '失败原因，成功忽略此参数',
        data: [ // 业务数据
            {
                url: '', // 图片路径(以 反斜杠 开头的绝对路径)
                alt: ''  // 图片描述，可为空
            },
            ...
        ]
    }

## 注册用户

url: /register

type: POST

input: 

    {
        email: '',
        userName: '',
        password: ''
    }

output:

    // 成功
    {
        code: 1,
        message: ''
    }

    // 失败
    {
        code: 0,
        message: '用户名已被使用' // 或者邮箱已被使用
    }

## 登录

url: login

type: POST

input:

    {
        userName: '',
        password: ''
    }

output:

    // 成功
    {
        code: 1,
        message: '',
        data: {
            uid: '',
            sign: ''
        }
    }
    
    // 失败
    {
        code: 0,
        message: '用户名或密码错误'
    }
